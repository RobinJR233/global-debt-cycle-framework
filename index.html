全球债务周期评估框架

主权债务不是静态的数字——它像经济本身一样，经历着可识别的膨胀、峰值与去杠杆周期。
这个框架帮你识别一个国家当前处于哪个阶段，以及接下来可能发生什么。

⚠️ 提前预警
Kaminsky 信号法监测 10 个领先指标，在债务危机爆发前 1-3 年发出信号。

🏛️ 投资与风控
主权 CDS 定价、新兴市场敞口管理、宏观对冲配置、地缘风险研判——债务周期是所有这些决策的底层逻辑。

src/data_ingestion/pipeline.pyPython
def fetch_country_data(country, start_year=2000, end_year=2026):
    raw = registry.fetch_all(country, start_year, end_year)
    merged = {}
    for year in range(start_year, end_year + 1):
        key = f"{country}_{year}"
        merged[key] = {"country": country, "year": year}
        for src, records in raw.items():
            for rec in records:
                if rec.get("year") == year:
                    for k, v in rec.items():
                        if k not in ("country","year","source"):
                            merged[key][k] = v
    return {"country": country, "data": list(merged.values())}

def compute_bis_credit_gap(credit_to_gdp, lam=400_000):
    """BIS Credit-to-GDP Gap = credit - HP_trend"""
    trend = _hp_filter(credit_to_gdp, lam)
    one_sided = _one_sided_hp_filter(credit_to_gdp, lam)
    gap = credit_to_gdp - trend
    gap[-3:] = credit_to_gdp[-3:] - one_sided[-3:]
    return gap

def compute_debt_snowball(debt_gdp, nom_growth, ir, pb):
    """delta_d = (r - g) * d + pb"""
    snowball = np.full(len(debt_gdp), np.nan)
    for t in range(1, len(debt_gdp)):
        r_g = ir[t] - nom_growth[t]
        snowball[t] = r_g * debt_gdp[t-1] + pb[t]
    return snowball

def classify_cycle_phase(bis_gap, debt_5y, cds, is_em=True):
    if bis_gap > 10:  return "peak"
    elif 5 < bis_gap <= 10: return "peak"
    elif -5 <= bis_gap <= 5:
        return "expansion" if debt_5y > 5 else "contraction"
    else: return "trough"

src/modeling/debt_sustainability.pyPython
def project_debt_snowball(d0, params, scenario, years=5):
    """IMF-style: delta_d = (r - g) * d + pb"""
    path = [d0]
    for t in range(1, years + 1):
        g = params["gdp_growth"][t] + scenario.gdp_growth_delta
        r = params["interest_rate"][t] + scenario.interest_rate_delta
        pb = params["primary_balance"][t] + scenario.pb_delta
        delta_d = (r - g) * path[t-1] + pb
        if scenario.fx_depreciation:
            delta_d += 0.4 * path[t-1] * scenario.fx_depreciation / 100
        path.append(max(0, path[t-1] + delta_d))
    return path

def monte_carlo_fan_chart(d0, params, years=5, n_sims=10000):
    paths = []
    for _ in range(n_sims):
        path = [d0]
        for t in range(1, years+1):
            g = params["gdp_growth"][t-1] + np.random.normal(0,1.0)
            r = params["interest_rate"][t-1] + np.random.normal(0,0.5)
            pb = params["primary_balance"][t-1] + np.random.normal(0,1.5)
            path.append(max(0, path[t-1] + (r-g)*path[t-1] + pb))
        paths.append(path)
    return {"median":[...], "p10":[...], "p90":[...]}

def assess_external_vulnerability(row):
    ext_debt = row.get("external_debt_pct_gdp", 40.0)
    reserves = row.get("fx_reserves_months_imports", 6.0)
    gfn = row.get("gfn_proxy", 50.0)
    reserves_pct_st = (reserves / 3.0) * 100
    gfn_ratio = (gfn / reserves_pct_st) * 100
    nii = row.get("nii_pct_gdp", 0.0)
    mismatch = abs(ext_debt - nii)
    fc_share = min(ext_debt / (ext_debt + 100), 0.6)
    stress_30pct = fc_share * 30.0 * ext_debt / 100.0
    risk_score = 0
    risk_score += 2 if gfn_ratio > 100 else 1 if gfn_ratio > 80 else 0

SCORING_FUNCTIONS = [
    _score_fiscal, _score_private_sector, _score_external,
    _score_market_pricing, _score_macro_context, _score_structural,
]

def compute_composite_score(row, prev=None):
    sub_scores = [fn(row) for fn in SCORING_FUNCTIONS]
    composite = sum(ss.weighted_score for ss in sub_scores)
    composite = max(0, min(100, round(composite, 1)))
    phase = classify_phase(composite)
    return CompositeScoreResult(composite_score=composite, phase=phase, ...)

# Green 70-100 | Blue 55-70 | Yellow 40-55 | Orange 25-40 | Red 0-25
# Score = sum(dim_score * weight)  weights sum = 1.0

def run_pipeline_for_country(country, start=2000, end=2026):
    # Stage 1: Data Ingestion
    raw = fetch_country_data(country, start, end)
    # Stage 2: Feature Engineering
    featured = engineer_features(raw["data"])
    # Stage 3: Early Warning Signals
    for row in featured:
        row.update(run_early_warning_signals(row, is_em=...))
    # Stage 4: Debt Sustainability Analysis
    dsm = run_debt_sustainability_analysis(country, featured[-1], featured)
    # Stage 5: External Vulnerability
    ext_vuln = assess_external_vulnerability(featured[-1])
    # Stage 6: Composite Scoring
    scored = [compute_composite_score(row, prev) for row in featured]
    return {"country": country, "years": scored,
            "dsm": dsm, "external": ext_vuln, ...}

# Usage: result = run_pipeline_for_country("USA")
